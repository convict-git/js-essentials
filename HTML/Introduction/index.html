         vector <int> pref_func (const string &s) { // O(|s|)
            int n = (int)s.size();
            vector <int> pi(n);
            pi[0] = 0;
            int sz = (int)s.size();
            for (int i = 1; i < sz; ++i) {
               int len_j = pi[i-1];
               int j = len_j - 1;
               while (len_j > 0 && s[j + 1] != s[i]) {
                  len_j = pi[j];
                  j = len_j - 1;
               }
               if (s[len_j] == s[i])
                  ++len_j;
               pi[i] = len_j;
            }
            return pi;
         }
